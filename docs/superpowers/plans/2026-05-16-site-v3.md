# Site v3 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Criar `site-v3/` com design moderno (top nav fixo, hero glassmorphism, paleta azul/vermelho) consumindo o `data.json` existente sem alterar `site/` nem `site-v2/`.

**Architecture:** Vanilla HTML + CSS + JS (ES2020, sem frameworks). Um `style.css` com design tokens completos. Um `app.js` com carregamento único de `data.json`, detecção de página e funções de render por seção. Cada HTML importa os mesmos dois arquivos.

**Tech Stack:** HTML5 · CSS3 (custom properties, grid, flexbox) · Vanilla JS ES2020 · Google Fonts (Sora + DM Sans + JetBrains Mono) · Python http.server para preview

---

## Mapa de arquivos

| Arquivo | Responsabilidade |
|---|---|
| `site-v3/style.css` | Design tokens, reset, nav, hero, cards, barras, tabelas, badges, mobile |
| `site-v3/app.js` | fetch data.json, initNav(), utilitários, render por página |
| `site-v3/index.html` | HTML shell da homepage |
| `site-v3/achados.html` | HTML shell do banco de achados |
| `site-v3/municipios.html` | HTML shell de por município |
| `site-v3/secoes.html` | HTML shell de por seção |
| `site-v3/defesa.html` | HTML shell de defesa do gestor |
| `site-v3/pareceres.html` | HTML shell de opinião do auditor |
| `site-v3/insights.html` | HTML shell de insights |
| `site-v3/sobre.html` | HTML shell sobre |
| `site-v3/data.json` | Cópia de site/data.json |

    renderIndex();

  const sitOrder = ['mantido','sanado_total','sanado_parcial','afastado','nao_consta'];
  const sitItems = sitOrder.filter(s => situacoes[s]).map(s => [SITUACAO_LABEL[s] || s, situacoes[s]]);
  document.getElementById('badge-total-sit').textContent = fmtN(total) + ' achados';

function renderAchadosPage(page) {
  _achadoPage = page;

    </tr>
    <tr class="accordion-detail" id="${uid}">
      <td colspan="6">

          <strong>Descrição:</strong> ${escHtml(a.descricao)}<br><br>
          ${a.recomendacao ? `<strong>Recomendação:</strong> ${escHtml(a.recomendacao)}<br><br>` : ''}
          ${a.determinacao ? `<strong>Determinação:</strong> ${escHtml(a.determinacao)}<br><br>` : ''}
          ${a.base_normativa ? `<strong>Base normativa:</strong> ${escHtml(a.base_normativa)}<br><br>` : ''}
          <strong>Seção:</strong> ${escHtml(a.secao)} · <strong>Processo:</strong> <span class="mono">${escHtml(a.numero_processo)}</span>
        </div>
      </td>
    </tr>`;
  }).join('');

  // paginação
  const total = FILTERED_ACHADOS.length;
  const pages = Math.ceil(total / ACHADOS_PER_PAGE);
  const pag = document.getElementById('achados-pag');
  if (pages <= 1) { pag.innerHTML = ''; return; }
  const btns = [];
  btns.push(`<button class="page-btn" onclick="renderAchadosPage(${page-1})" ${page<=1?'disabled':''}>‹</button>`);
  for (let p = Math.max(1,page-2); p <= Math.min(pages,page+2); p++) {
    btns.push(`<button class="page-btn${p===page?' active':''}" onclick="renderAchadosPage(${p})">${p}</button>`);
  }
  btns.push(`<button class="page-btn" onclick="renderAchadosPage(${page+1})" ${page>=pages?'disabled':''}>›</button>`);
  btns.push(`<span class="page-info">${fmtN(total)} resultados</span>`);
  pag.innerHTML = btns.join('');
}

function toggleAccordion(uid) {
  const detail = document.getElementById(uid);
  const row = document.querySelector(`[data-uid="${uid}"]`);
  const isOpen = detail.classList.contains('open');
  // fechar todos
  document.querySelectorAll('.accordion-detail.open').forEach(el => {
    el.classList.remove('open');
    document.querySelector(`[data-uid="${el.id}"]`)?.classList.remove('open');
  });
  if (!isOpen) { detail.classList.add('open'); row?.classList.add('open'); }
}

function renderMunicipios() {

function renderInsights() {
  const ag = DATA.agregacoes;
  const t  = ag.totais;
  const pctMantido = Math.round(ag.por_situacao.mantido / t.achados * 100);
  const pctSaneado = Math.round(((ag.por_situacao.sanado_total||0)+(ag.por_situacao.sanado_parcial||0)) / t.achados * 100);
  const topMuni = Object.entries(ag.por_municipio).sort((a,b)=>b[1]-a[1])[0];
  const topTipo = Object.entries(ag.por_tipo).sort((a,b)=>b[1]-a[1])[0];
  const topSecao = Object.entries(ag.por_secao).sort((a,b)=>b[1]-a[1])[0];
  const comDefPct = Math.round(t.com_defesa / t.achados * 100);
  const defMantido = ag.defesa_x_situacao?.com_defesa?.mantido || 0;
  const defMantidoPct = t.com_defesa ? Math.round(defMantido/t.com_defesa*100) : 0;
  const irregular = DATA.relatorios.filter(r => r.opiniao_auditoria === 'Irregular').length;

  const insights = [
    { cor: 'red',   titulo: `${pctMantido}% dos achados foram mantidos`,
      texto: `De ${fmtN(t.achados)} achados analisados, ${fmtN(ag.por_situacao.mantido)} permanecem como irregularidade após o contraditório. Isso indica que a maioria das falhas identificadas não foi corrigida a tempo.` },
    { cor: 'blue',  titulo: `${pctSaneado}% foram saneados`,
      texto: `${fmtN((ag.por_situacao.sanado_total||0)+(ag.por_situacao.sanado_parcial||0))} achados foram total ou parcialmente corrigidos, demonstrando que o processo de contraditório produz resultados em parcela relevante dos casos.` },
